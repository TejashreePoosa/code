#include <stdio.h>
int recgcd(int x, int y);
int nonrecgcd(int x, int y);
int main()
{
 int a, b, c, d;
 printf("Enter two numbers ");
 scanf("%d %d", &a, &b);
 c = recgcd(a, b);
 printf("\nGCD of %d,%d using recursion is %d", a,b,c);
 d = nonrecgcd(a, b);
 printf("\nGCD of %d,%d using nonrecursion is %d", a,b,d);
}
int recgcd(int x, int y)
{
 if(y == 0)
 {
 return(x);
 }
 else
 {
 return(recgcd(y, x % y));
 }
}
int nonrecgcd(int x, int y)

{
 int z;
 while(x % y != 0)
 {
 z = x % y;
 x = y;
 y = z;
 }
 return(y);
}
